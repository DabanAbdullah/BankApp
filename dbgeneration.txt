CREATE VIEW [dbo].[incommingtransaction]
AS
SELECT dbo.Accounts.AccountId, dbo.Transactioncounterpart.CounterpartyIBAN, dbo.Transactioncounterpart.TransactionType, dbo.Transactioncounterpart.TransactionId, dbo.Accounts.CustomerId
FROM     dbo.Transactioncounterpart INNER JOIN
                  dbo.Accounts ON dbo.Transactioncounterpart.CounterpartyIBAN = dbo.Accounts.IBAN
WHERE  (dbo.Transactioncounterpart.TransactionType = 3)
GO

CREATE VIEW [dbo].[transactionview2]
AS
SELECT dbo.incommingtransaction.TransactionId, 'incoming' AS name, dbo.incommingtransaction.AccountId, dbo.Transactions.TransactionDate, dbo.Transactions.Amount, dbo.Transactions.Purpose, 
                  dbo.Transactioncounterpart.yourIBAN AS counterpartyIBAN, dbo.Transactioncounterpart.CounterpartyIBAN AS yourIBAN, dbo.Transactions.IsDeleted
FROM     dbo.TransactionTypes INNER JOIN
                  dbo.Accounts INNER JOIN
                  dbo.incommingtransaction ON dbo.Accounts.AccountId = dbo.incommingtransaction.AccountId ON dbo.TransactionTypes.TransactionTypeId = dbo.incommingtransaction.TransactionType INNER JOIN
                  dbo.Transactions ON dbo.incommingtransaction.TransactionId = dbo.Transactions.TransactionId INNER JOIN
                  dbo.Transactioncounterpart ON dbo.incommingtransaction.TransactionId = dbo.Transactioncounterpart.TransactionId
GO

CREATE VIEW [dbo].[transactionview]

CREATE VIEW [dbo].[finalTransaction]
AS
SELECT ROW_NUMBER() OVER (ORDER BY 
                      (SELECT NULL)) AS Id, /* primary key-like column*/ *
FROM     (SELECT *
                  FROM      dbo.transactionview
                  UNION
                  SELECT *
                  FROM     dbo.transactionview2) AS Combined;

/****** Object:  View [dbo].[AccountBalance]    Script Date: 10/25/2025 9:13:26 PM ******/
SET ANSI_NULLS ON
GO
SET QUOTED_IDENTIFIER ON
GO
CREATE VIEW [dbo].[AccountBalance]
AS
SELECT TOP (100) PERCENT a.AccountId, a.OpeningBalance + ISNULL(SUM(CASE WHEN LOWER(ft.Name) IN ('incoming', 'deposit') THEN ft.Amount WHEN LOWER(ft.Name) IN ('withdrawal', 'transfer') THEN - ft.Amount ELSE 0 END), 0) 
                  AS TotalBalance
FROM     dbo.Accounts AS a LEFT OUTER JOIN
                  dbo.finalTransaction AS ft ON ft.AccountId = a.AccountId
GROUP BY a.AccountId, a.OpeningBalance
ORDER BY a.AccountId
GO
